异步方法

使用 async await 处理异步流程

处理异步方法，必须使用 try catch捕获异常

function onePromise() {
    return Promise.reolve('Hello Async')
}

async myFunc() {
    try {
        const result = await onePromise()
        console.log(result)
    }
    catch(e) {
        console.error(e)
    }
}

复杂的异常处理场景，考虑使用 Promise

异步当作同步方法来执行，async 可读性高

复杂异常场景的处理，Promise 复杂度更低

/**
 * async 处理复杂的异常情况
 */
function onePromise() {
    throw new Error('something wrong')
}

function failHandler() {
    throw new Error('I can not finish it')
}

async function myFunc() {
    try {
        const result = await onePromise()
        console.log('success')
    }
    catch(e) {
        console.error(e)
        try {
            await failHandler()
            console.log('no error')
        }
        catch(err) {
            console.error(err)
        }
    }
}

/**
 * Promise 处理复杂的异常情况
 */
function onePromise() {
    throw new Error('something wrong')
}

function failHandler() {
    throw new Error('I can not finish it')
}

function myFunc() {
    onePromise()
    .then(result => {
        console.log('success')
    })
    .catch(e => {
        return failHandler()
    })
    .then(result => {
        console.log('no error')
    })
    .catch(e => {
        console.error(e)
    })
}

if 中 return 的使用

如果这两组代码是互斥关系，建议使用 if & else

function getTitle() {
    if (window.dev) {
        reutrn 'dev title'
    }
    else {
        return 'real title'
    }
}

如果是要首先排除某些特殊的错误情况，建议使用 return

function doSomething(response) {
    // 提前判断异常情况
    if (response.err) return
    // do something
}

map结构 vs switch

map 结构清晰简洁，优先使用 map 方法处理分支场景

// 这是一个 enum
const StudentType = {
    TYPE1: 1,
    TYPE2: 2,
    TYPE3: 3
}

const msgMap = {
    [StudentType.TYPE1]: 'hello Vue',
    [StudentType.TYPE2]: 'hello Javascript',
    [StudentType.TYPE3]: 'hello Html',
}

// 使用 map 方法返回 msg
function getMsgByStudentType(type) {
    return msgMap[type]
}
